void setup() {
  // put your setup code here, to run once:
  pinMode(INPUT, A0);
  pinMode(OUTPUT, 9);
  delay(150);
  Motor.attach(9); // Motor- PWM and Pin 9 ausgeben
  Serial.begin(115200);
}
